def analyze_cycle_history(history):
    """Mine successful fixes from cycle history."""
    techniques = []
    for i, cycle in enumerate(history):
        if i > 0:
            score_delta = cycle["score"] - history[i-1]["score"]
            if score_delta > 0:
                fix_applied = get_fix_from_cycle(cycle)
                techniques.append({
                    "technique": fix_applied,
                    "impact_delta": score_delta,
                    "cycle": i,
                    "reliability": 1.0 if score_delta > 0.5 else 0.5
                })
    return sorted(techniques, key=lambda x: x["impact_delta"], reverse=True)

def validate_render_batch(render_dir, scene_names, threshold_brightness=10):
    """
    CRITICAL VALIDATION GATE
    Checks:
    1. Brightness: mean(RGB) > threshold (default 10)
    2. Color variance: std(R), std(G), std(B) > 5
    3. Geometry: Sobel edge detection, density > 0.02
    4. File size: > 100KB for 1920x1080
    5. Format: PIL can load without error
    
    Returns early if ANY check fails.
    """
    failures = {}
    metrics = {}
    
    for scene_name in scene_names:
        render_file = f"{render_dir}/{scene_name}.png"
        
        # Load image
        img = Image.open(render_file)
        arr = np.array(img)
        
        # Test 1: Brightness
        brightness = np.mean(arr[:,:,:3])
        if brightness <= threshold_brightness:
            failures[scene_name] = "TOO_DARK"
            metrics[scene_name] = {"brightness": brightness}
            continue
        
        # Test 2: Color variance
        r_var = np.std(arr[:,:,0])
        g_var = np.std(arr[:,:,1])
        b_var = np.std(arr[:,:,2])
        color_var = (r_var + g_var + b_var) / 3
        
        if color_var < 5:
            failures[scene_name] = "MONOCHROME"
            metrics[scene_name] = {"color_variance": color_var}
            continue
        
        # Test 3: Geometry (Sobel edges)
        gray = cv2.cvtColor(arr.astype(np.uint8), cv2.COLOR_RGB2GRAY)
        sobelx = cv2.Sobel(gray, cv2.CV_64F, 1, 0, ksize=3)
        sobely = cv2.Sobel(gray, cv2.CV_64F, 0, 1, ksize=3)
        edges = np.sqrt(sobelx**2 + sobely**2)
        edge_density = np.sum(edges > 10) / edges.size
        
        if edge_density < 0.02:
            failures[scene_name] = "NO_GEOMETRY"
            metrics[scene_name] = {"edge_density": edge_density}
            continue
        
        # Test 4: File size
        file_size = os.path.getsize(render_file) / 1024  # KB
        if file_size < 100:
            failures[scene_name] = "FILE_TOO_SMALL"
            metrics[scene_name] = {"file_size_kb": file_size}
            continue
        
        # All tests passed
        metrics[scene_name] = {
            "brightness": brightness,
            "color_variance": color_var,
            "edge_density": edge_density,
            "file_size_kb": file_size,
            "status": "PASS"
        }
    
    valid = len(failures) == 0
    return valid, failures, metrics

Goal: Score renders on 10-point scale with top 5 problems
Input: Render files, scene names

def run_harsh_audit(render_results, scene_list):
    """
    Evaluate forensic animation quality.
    Dimensions:
    - Lighting quality (shadow detail, fill ratio, rim definition)
    - Geometry clarity (sharp edges, no clipping, scale visible)
    - Color accuracy (neutral white balance, forensic realism)
    - Composition (framing, forensic evidence centered)
    - Technical specs (noise, alias, resolution clarity)
    
    Top 5 problems ranked by severity and prevalence.
    """
    audit = {
        "overall_score": 0,
        "top_5_problems": [],
        "scene_scores": {},
        "timestamp": datetime.now().isoformat()
    }
    
    scene_scores = []
    all_problems = []
    
    for scene_name in scene_list:
        render_file = render_results[scene_name]
        
        # Evaluate this scene
        score, problems = evaluate_scene(render_file, scene_name)
        scene_scores.append(score)
        audit["scene_scores"][scene_name] = score
        
        for problem in problems:
            all_problems.append({
                "issue": problem["issue"],
                "severity": problem["severity"],
                "scenes": [scene_name],
                "explanation": problem["explanation"]
            })
    
    # Consolidate: merge same problems across scenes
    consolidated = consolidate_problems(all_problems)
    
    # Rank by severity * prevalence
    for problem in consolidated:
        problem["rank_score"] = problem["severity"] * len(problem["scenes"])
    
    consolidated.sort(key=lambda x: x["rank_score"], reverse=True)
    
    # Top 5
    audit["top_5_problems"] = [
        {
            "rank": i + 1,
            "issue": p["issue"],
            "severity": p["severity"],
            "scenes": p["scenes"],
            "explanation": p["explanation"]
        }
        for i, p in enumerate(consolidated[:5])
    ]
    
    # Overall score: average of scene scores
    audit["overall_score"] = np.mean(scene_scores)
    
    return audit

def consolidate_research_findings(research_findings):
    """
    Merge results from YouTube, forum, and translation agents.
    Deduplicate. Rank by confidence and applicability.
    Return top 5 techniques ready to code.
    """
    all_techniques = []
    
    for finding in research_findings:
        all_techniques.extend(finding.get("techniques", []))
    
    # Deduplicate by description similarity
    deduped = []
    for tech in all_techniques:
        is_dup = False
        for existing in deduped:
            if jaccard_similarity(tech["title"], existing["title"]) > 0.7:
                existing["source_count"] += 1
                is_dup = True
                break
        if not is_dup:
            tech["source_count"] = 1
            deduped.append(tech)
    
    # Rank by source_count, confidence, applicability
    deduped.sort(key=lambda x: (x["source_count"], x["confidence"]), reverse=True)
    
    return deduped[:5]

def translate_findings_to_code(findings, target_format="bpy_mcp_calls", max_blocks=5):
    """
    Read research findings. For each:
    1. Identify the technique (e.g., "use HDRI + world background")
    2. Extract pseudocode or steps
    3. Translate to bpy API calls
    4. Format as MCP-compatible function
    5. Test syntax
    """
    code_blocks = []
    
    for finding in findings:
        # Parse technique
        technique_name = finding["title"]
        technique_desc = finding["description"]
        
        # Generate bpy code
        bpy_code = generate_bpy_code(technique_name, technique_desc)
        
        # Validate syntax
        try:
            compile(bpy_code, '<string>', 'exec')
            code_blocks.append({
                "technique": technique_name,
                "code": bpy_code,
                "status": "valid"
            })
        except SyntaxError as e:
            print(f"Syntax error in {technique_name}: {e}")
    
    return code_blocks[:max_blocks]
